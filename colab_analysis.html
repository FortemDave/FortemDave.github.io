# Cleaning function
def clean_df_colab(df):
    df = df.rename(columns=lambda s: s.strip())
    sel = (df['State'] == 0) & (df['Distt.'] == 0) & (df['Area Name'].str.strip().str.lower() == 'india')
    ndf = df[sel].copy()
    ndf['Age'] = ndf['Age'].astype(str).str.strip()
    def to_int_series(ndf, name):
        candidates = [c for c in ndf.columns if c.strip()==name]
        c = candidates[0] if candidates else None
        if c is None:
            return pd.Series([pd.NA]*len(ndf))
        return pd.to_numeric(ndf[c].astype(str).str.replace(',', '').str.strip(), errors='coerce')
    clean = pd.DataFrame()
    clean['Age'] = ndf['Age']
    clean['Total Persons'] = to_int_series(ndf, 'Total Persons')
    clean['Total Males'] = to_int_series(ndf, 'Total Males')
    clean['Total Females'] = to_int_series(ndf, 'Total Females')
    clean['Rural Persons'] = to_int_series(ndf, 'Rural Persons')
    clean['Urban Persons'] = to_int_series(ndf, 'Urban Persons')
    clean['AgeNum'] = clean['Age'].apply(lambda a: int(a) if a.isdigit() else pd.NA)
    return clean

clean = clean_df_colab(df)
clean.to_csv('outputs/india_age_national.csv', index=False)
print('Cleaned rows:', len(clean))

# Line plot with LOWESS and peak annotation
plt.figure(figsize=(10,6))
plt.plot(clean2['AgeNum'], clean2['Total Persons'], marker='o', markersize=3, alpha=0.6, label='raw')
try:
    sm = lowess(clean2['Total Persons'], clean2['AgeNum'], frac=0.1)
    plt.plot(sm[:,0], sm[:,1], color='red', linewidth=1.5, label='LOWESS')
except Exception:
    pass
peak_idx = clean2['Total Persons'].idxmax()
peak_age = int(clean2.loc[peak_idx,'AgeNum'])
peak_val = int(clean2.loc[peak_idx,'Total Persons'])
plt.annotate(f'Peak age {peak_age}\n{peak_val:,}', xy=(peak_age, peak_val), xytext=(peak_age+5, peak_val*0.9), arrowprops=dict(arrowstyle='->'))
plt.title('Population by single-year age (India, 2001)')
plt.xlabel('Age')
plt.ylabel('Total Persons')
plt.grid(True, linestyle=':')
plt.legend()
plt.tight_layout()

mask = x.notna() & y.notna()
xp = x[mask]
yp = y[mask]
try:
    m,b = np.polyfit(xp, yp,1)
    plt.plot([0,mx],[b,m*mx+b], color='green', label=f'fit y={m:.3f}x+{b:.0f}')
